// Simple job data
let job = {
  id: 1,
  title: "Friday Night Shift",
  bar: "Midnight Lounge",
  pay: "$250",
  hours: "8pm – 2am",
  location: "Downtown",
  notes: "Casual dress code",
  matched: false
};

const jobCard = document.getElementById("job-card");
const matchList = document.getElementById("match-list");
const declineBtn = document.getElementById("decline");
const acceptBtn = document.getElementById("accept");

// Accept or Decline logic
acceptBtn.addEventListener("click", () => handleSwipe("right"));
declineBtn.addEventListener("click", () => handleSwipe("left"));

function handleSwipe(direction) {
  if(direction === "right") {
    job.matched = true;
    const li = document.createElement("li");
    li.textContent = `${job.title} at ${job.bar} ✅ Matched`;
    matchList.appendChild(li);
    alert("You accepted the job!");
  } else {
    alert("You declined the job ❌");
  }
}

// Messaging
const msgInput = document.getElementById("msg-input");
const sendBtn = document.getElementById("send-msg");
const msgList = document.getElementById("msg-list");

sendBtn.addEventListener("click", () => {
  const text = msgInput.value.trim();
  if(text) {
    const li = document.createElement("li");
    li.textContent = text;
    msgList.appendChild(li);
    msgInput.value = "";
  }
});

// Ratings
const ratingSelect = document.getElementById("rating");
const ratingComment = document.getElementById("rating-comment");
const submitRating = document.getElementById("submit-rating");
const ratingsList = document.getElementById("ratings-list");

submitRating.addEventListener("click", () => {
  const stars = ratingSelect.value;
  const comment = ratingComment.value.trim();
  if(stars) {
    const li = document.createElement("li");
    li.textContent = `${"⭐".repeat(stars)} ${comment}`;
    ratingsList.appendChild(li);
    ratingSelect.value = "";
    ratingComment.value = "";
    alert("Rating submitted!");
  } else {
    alert("Select a rating first!");
  }
});
